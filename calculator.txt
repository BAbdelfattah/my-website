<body>
    <!-- Calculator container, hidden by default -->
    <div id="calculatorContainer" class="calculator-container" style="display: none;">
        <h2>Food Safety Calculator</h2>
        <label for="weight">Enter Weight (in grams):</label>
        <input type="number" id="weight" name="weight" min="0" step="1" required>
        <br><br>
        <button onclick="calculateFoodSafety()">Calculate</button>
        <br><br>
        <p id="result"></p>
    </div>

    <script>
        // JavaScript to show the calculator once on page load
        document.addEventListener('DOMContentLoaded', function() {
            // Show the calculator container
            document.getElementById('calculatorContainer').style.display = 'block';

            // Optional: Use localStorage to track if the calculator has been shown
            if (localStorage.getItem('calculatorShown') !== 'true') {
                // Set item in localStorage to indicate calculator has been shown
                localStorage.setItem('calculatorShown', 'true');
            }
        });

        // Function to calculate food safety based on weight
        function calculateFoodSafety() {
            // Get input value
            var weight = document.getElementById('weight').value;

            // Perform calculation (example: check if weight is safe)
            var resultMessage = "";
            if (weight > 1000) {
                resultMessage = "This amount may be unsafe. Please handle with caution.";
            } else {
                resultMessage = "This amount is safe for handling.";
            }

            // Display result
            document.getElementById('result').innerHTML = resultMessage;
        }
    </script>
